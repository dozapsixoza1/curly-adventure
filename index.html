JavaScript — CryptoPlay SDK
// Install: npm install @cryptoplay/sdk
import { CryptoPlay } from '@cryptoplay/sdk';

// Initialize with your game ID const cp = new CryptoPlay({ gameId: 'your-game-id', network: 'mainnet' });

// Authenticate via Telegram const user = await cp.auth.loginWithTelegram();

// Process in-game payment await cp.payments.charge({ amount: '0.5', // TON description: 'Buy Magic Sword NFT' });

// Mint NFT to player await cp.nft.mint({ to: user.wallet, metadata: { name: 'Magic Sword', rarity: 'legendary' } });
